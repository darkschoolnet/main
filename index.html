Games-Tabelle ───────────────────────────────────────────── CREATE TABLE IF NOT EXISTS dsn_games ( id UUID DEFAULT gen_random_uuid() PRIMARY KEY, name TEXT NOT NULL, url TEXT NOT NULL, image_base64 TEXT, -- quadratisches Bild, client-seitig auf 200x200 komprimiert emoji TEXT DEFAULT '🎮', -- Fallback wenn kein Bild hochgeladen accent_color TEXT DEFAULT '#5b8cff', glow_color TEXT DEFAULT 'rgba(91,140,255,0.12)', type_label TEXT DEFAULT 'APP', description TEXT DEFAULT '', sort_order INTEGER DEFAULT 0, active BOOLEAN DEFAULT true, created_at TIMESTAMPTZ DEFAULT NOW() ); ALTER TABLE dsn_games ENABLE ROW LEVEL SECURITY; -- Öffentlicher Lesezugriff nur auf aktive Games (kein salt/hash etc. hier) DROP POLICY IF EXISTS "Games lesbar" ON dsn_games; CREATE POLICY "Games lesbar" ON dsn_games FOR SELECT USING (active = true); -- Kein direkter Schreibzugriff für Anon — nur über Admin-RPCs DROP POLICY IF EXISTS "Games kein Anon Write" ON dsn_games; CREATE POLICY "Games kein Anon Write" ON dsn_games FOR ALL USING (false) WITH CHECK (false); -- ── 2. Initiale Games eintragen ────────────────────────────────── -- (Nur wenn Tabelle noch leer ist) INSERT INTO dsn_games (name, url, emoji, accent_color, glow_color, type_label, description, sort_order, active) SELECT * FROM (VALUES ('Block Blast', 'https://darkschoolnet.github.io/blockblast', '🟦', '#5b8cff', 'rgba(91,140,255,0.12)', 'SPIEL · PUZZLE', 'Platziere Blöcke, räume Reihen ab und erreich den Highscore. Der Klassiker auf dem Schul-iPad.', 1, true), ('Crossy Roads', 'https://darkschoolnet.github.io/crossyroad', '🐔', '#ffaa33', 'rgba(255,170,51,0.1)', 'SPIEL · ARCADE', 'Spring über Straßen, Gleise und Flüsse. Wie weit kommst du? Simpel, direkt, süchtig machend.', 2, true), ('Chats', 'https://darkschoolnet.github.io/chat', '💬', '#44cc88', 'rgba(68,204,136,0.1)', 'APP · KOMMUNIKATION', 'Schreib direkt mit Mitschülern im DSN-Netz. Kein Lehrerzugriff, kein Drama.', 3, true) ) AS v(name, url, emoji, accent_color, glow_color, type_label, description, sort_order, active) WHERE NOT EXISTS (SELECT 1 FROM dsn_games); -- ════════════════════════════════════════════════════════════════ -- ÖFFENTLICHE RPC — Games laden (kein Admin nötig) -- ════════════════════════════════════════════════════════════════ CREATE OR REPLACE FUNCTION dsn_get_games() RETURNS JSONB LANGUAGE plpgsql SECURITY DEFINER AS $$ BEGIN RETURN ( SELECT COALESCE(jsonb_agg( jsonb_build_object( 'id', g.id, 'name', g.name, 'url', g.url, 'image_base64', g.image_base64, 'emoji', g.emoji, 'accent_color', g.accent_color, 'glow_color', g.glow_color, 'type_label', g.type_label, 'description', g.description, 'sort_order', g.sort_order ) ORDER BY g.sort_order ASC, g.created_at ASC ), '[]'::JSONB) FROM dsn_games g WHERE g.active = true ); END; $$; -- ════════════════════════════════════════════════════════════════ -- ADMIN RPCs — alle mit Hash-Verifikation -- ════════════════════════════════════════════════════════════════ -- Alle Games laden (inkl. inaktive, für Admin) CREATE OR REPLACE FUNCTION dsn_admin_get_games(p_admin_hash TEXT) RETURNS JSONB LANGUAGE plpgsql SECURITY DEFINER AS $$ BEGIN IF NOT EXISTS(SELECT 1 FROM dsn_config WHERE key='admin_password' AND value->>'hash'=p_admin_hash) THEN RETURN NULL; END IF; RETURN ( SELECT COALESCE(jsonb_agg( jsonb_build_object( 'id', g.id, 'name', g.name, 'url', g.url, 'image_base64', g.image_base64, 'emoji', g.emoji, 'accent_color', g.accent_color, 'glow_color', g.glow_color, 'type_label', g.type_label, 'description', g.description, 'sort_order', g.sort_order, 'active', g.active, 'created_at', g.created_at ) ORDER BY g.sort_order ASC, g.created_at ASC ), '[]'::JSONB) FROM dsn_games g ); END; $$; -- Game hinzufügen CREATE OR REPLACE FUNCTION dsn_admin_add_game( p_admin_hash TEXT, p_name TEXT, p_url TEXT, p_emoji TEXT, p_image_base64 TEXT, p_accent_color TEXT, p_glow_color TEXT, p_type_label TEXT, p_description TEXT, p_sort_order INTEGER ) RETURNS JSONB LANGUAGE plpgsql SECURITY DEFINER AS $$ DECLARE v_id UUID; BEGIN IF NOT EXISTS(SELECT 1 FROM dsn_config WHERE key='admin_password' AND value->>'hash'=p_admin_hash) THEN RETURN jsonb_build_object('error','unauthorized'); END IF; INSERT INTO dsn_games(name,url,emoji,image_base64,accent_color,glow_color,type_label,description,sort_order,active) VALUES(p_name,p_url,p_emoji,NULLIF(p_image_base64,''),p_accent_color,p_glow_color,p_type_label,p_description,p_sort_order,true) RETURNING id INTO v_id; RETURN jsonb_build_object('success',true,'id',v_id); END; $$; -- Game aktualisieren CREATE OR REPLACE FUNCTION dsn_admin_update_game( p_admin_hash TEXT, p_game_id UUID, p_name TEXT, p_url TEXT, p_emoji TEXT, p_image_base64 TEXT, p_accent_color TEXT, p_glow_color TEXT, p_type_label TEXT, p_description TEXT, p_sort_order INTEGER, p_active BOOLEAN ) RETURNS BOOLEAN LANGUAGE plpgsql SECURITY DEFINER AS $$ BEGIN IF NOT EXISTS(SELECT 1 FROM dsn_config WHERE key='admin_password' AND value->>'hash'=p_admin_hash) THEN RETURN false; END IF; UPDATE dsn_games SET name = p_name, url = p_url, emoji = p_emoji, image_base64 = CASE WHEN p_image_base64 = '__keep__' THEN image_base64 ELSE NULLIF(p_image_base64,'') END, accent_color = p_accent_color, glow_color = p_glow_color, type_label = p_type_label, description = p_description, sort_order = p_sort_order, active = p_active WHERE id = p_game_id; RETURN FOUND; END; $$; -- Game löschen CREATE OR REPLACE FUNCTION dsn_admin_delete_game(p_admin_hash TEXT, p_game_id UUID) RETURNS BOOLEAN LANGUAGE plpgsql SECURITY DEFINER AS $$ BEGIN IF NOT EXISTS(SELECT 1 FROM dsn_config WHERE key='admin_password' AND value->>'hash'=p_admin_hash) THEN RETURN false; END IF; DELETE FROM dsn_games WHERE id = p_game_id; RETURN FOUND; END; $$; -- ════════════════════════════════════════════════════════════════ -- EXECUTE-RECHTE -- ════════════════════════════════════════════════════════════════ GRANT EXECUTE ON FUNCTION dsn_get_games() TO anon; GRANT EXECUTE ON FUNCTION dsn_admin_get_games(TEXT) TO anon; GRANT EXECUTE ON FUNCTION dsn_admin_add_game(TEXT,TEXT,TEXT,TEXT,TEXT,TEXT,TEXT,TEXT,TEXT,INTEGER) TO anon; GRANT EXECUTE ON FUNCTION dsn_admin_update_game(TEXT,UUID,TEXT,TEXT,TEXT,TEXT,TEXT,TEXT,TEXT,TEXT,INTEGER,BOOLEAN) TO anon; GRANT EXECUTE ON FUNCTION dsn_admin_delete_game(TEXT,UUID) TO anon;
